  144.     function parseDir($dir,$options = array())
    145. 

  145.     {
    146. 

  146.         $files = array();
    147. 

  147.         $latest_version = $this->latest_version;
    148. 

  148.         $extensions = array();
    149. 

  149.         $constants = array();
    150. 

  150.         $ignored = array();
    151. 

  151.         $default_options = array('file_ext' => array('php','php4','inc','phtml'), 'recurse_dir' => true, 'debug' => false, 'ignore_files' => array(), 'ignore_dirs' => array());
    152. 

  152.         $options = array_merge($default_options,$options);
    153. 

  153.         
    154. 

  154.         if(is_dir($dir) && is_readable($dir)) {
    155. 

  155.             if($dir{strlen($dir)-1} == '/' || $dir{strlen($dir)-1} == '\\') {
    156. 

  156.                 $dir = substr($dir,0,-1);
    157. 

  157.             }
    158. 

  158.             array_map('strtolower', $options['file_ext']);
    159. 

  159.             array_map('strtolower', $options['ignore_files']);
    160. 

  160.             array_map('strtolower', $options['ignore_dirs']);
    161. 

  161.             $files_raw = $this->_fileList($dir,$options);
    162. 

  162.             foreach($files_raw as $file) {
    163. 

  163.                 if(in_array(strtolower($file),$options['ignore_files'])) {
    164. 

  164.                     $ignored[] = $file;
    165. 

  165.                     continue;
    166. 

  166.                 }
    167. 

  167.                 $file_info = pathinfo($file);
    168. 

  168.                 if (isset($file_info['extension']) && in_array(strtolower($file_info['extension']),$options['file_ext'])) {
    169. 

  169.                     $tokens = $this->_tokenize($file);
    170. 

  170.                     if ($tokens != false) {
    171. 

  171.                         $files[$file] = $this->_parseTokens($tokens,$options);
    172. 

  172.                     } else {
    173. 

  173.                         return false;
    174. 

  174.                     }
    175. 

  175.                 }
    176. 

  176.             }
    177. 

  177.             foreach($files as $file) {
    178. 

  178.                 $cmp = version_compare($latest_version,$file['version']);
    179. 

  179.                 if ((int)$cmp === -1) {
    180. 

  180.                     $latest_version = $file['version'];
    181. 

  181.                 }
    182. 

  182.                 foreach($file['extensions'] as $ext) {
    183. 

  183.                     if(!in_array($ext,$extensions)) {
    184. 

  184.                         $extensions[] = $ext;
    185. 

  185.                     }
    186. 

  186.                 }
    187. 

  187.                 foreach ($file['constants'] as $const) {
    188. 

  188.                     if(!in_array($const,$constants)) {
    189. 

  189.                         $constants[] = $const;
    190. 

  190.                     }
    191. 

  191.                 }
    192. 

  192.             }
    193. 

  193.  
    194. 

  194.             if (sizeof($files) < 1) {
    195. 

  195.                 return false;
    196. 

  196.             }
    197. 

  197.  
    198. 

  198.             $files['constants'] = $constants;
    199. 

  199.             $files['extensions'] = $extensions;
    200. 

  200.             $files['version'] = $latest_version;
    201. 

  201.             $files['ignored_files'] = $ignored;
    202. 

  202.  
    203. 

  203.             return array_reverse($files);
    204. 

  204.         } else {
    205. 

  205.             return false;
    206. 

  206.         }
    207. 

  207.     }

  248.     function parseArray($files,$options = array()) {
    249. 

  249.         $latest_version = $this->latest_version;
    250. 

  250.         $extensions = array();
    251. 

  251.         $constants = array();
    252. 

  252.         $options = array_merge(array('file_ext' => array('php','php4','inc','phtml'), 'is_string' => false,'debug' => false, 'ignore_files' => array()),$options);
    253. 

  253.         $options['ignore_files'] = array_map("strtolower",$options['ignore_files']);
    254. 

  254.         foreach($files as $file) {                
    255. 

  255.             if ($options['is_string'] == false) {
    256. 

  256.                 $pathinfo = pathinfo($file);
    257. 

  257.                 if (!in_array(strtolower($file),$options['ignore_files']) && in_array($pathinfo['extension'],$options['file_ext'])) {
    258. 

  258.                     $tokens = $this->_tokenize($file,$options['is_string']);
    259. 

  259.                     if ($tokens != false) {
    260. 

  260.                         $files_parsed[$file] = $this->_parseTokens($tokens,$options);
    261. 

  261.                     } else {
    262. 

  262.                         $files_parsed[$file] = false;
    263. 

  263.                     }
    264. 

  264.                 } else {
    265. 

  265.                     $ignored[] = $file;
    266. 

  266.                 }
    267. 

  267.             } else {
    268. 

  268.                 $tokens = $this->_tokenize($file,$options['is_string']);
    269. 

  269.                 if ($tokens != false) {
    270. 

  270.                     $files_parsed[] = $this->_parseTokens($tokens,$options);
    271. 

  271.                 } else {
    272. 

  272.                     $files_parsed[] = false;
    273. 

  273.                 }
    274. 

  274.             }
    275. 

  275.         }
    276. 

  276.  
    277. 

  277.         foreach($files_parsed as $file) {
    278. 

  278.             if ($file != false) {
    279. 

  279.                 $cmp = version_compare($latest_version,$file['version']);
    280. 

  280.                 if ((int)$cmp === -1) {
    281. 

  281.                     $latest_version = $file['version'];
    282. 

  282.                 }
    283. 

  283.                 foreach($file['extensions'] as $ext) {
    284. 

  284.                     if(!in_array($ext,$extensions)) {
    285. 

  285.                         $extensions[] = $ext;
    286. 

  286.                     }
    287. 

  287.                 }
    288. 

  288.                 foreach($file['constants'] as $const) {
    289. 

  289.                     if(!in_array($const,$constants)) {
    290. 

  290.                         $constants[] = $const;
    291. 

  291.                     }
    292. 

  292.                 }
    293. 

  293.             }
    294. 

  294.         }
    295. 

  295.  
    296. 

  296.         $files_parsed['constants'] = $constants;
    297. 

  297.         $files_parsed['extensions'] = $extensions;
    298. 

  298.         $files_parsed['version'] = $latest_version;
    299. 

  299.         $files_parsed['ignored_files'] =  isset($ignored) ? $ignored : array();
    300. 

  300.         return array_reverse($files_parsed);
    301.

  303.     /**
    304. 

  304.      * Parse the given Tokens
    305. 

  305.      *
    306. 

  306.      * The tokens are those returned by
    307. 

  307.      * token_get_all() which is nicely
    308. 

  308.      * wrapped in PHP_CompatInfo::_tokenize
    309. 

  309.      *
    310. 

  310.      * @param array $tokens Array of PHP Tokens
    311. 

  311.      * @param boolean $debug Show Extra Output
    312. 

  312.      * @access private
    313. 

  313.      * @return array 
    314.

    315. 

  315.  
    316. 

  316.     function _parseTokens($tokens, $options)
    317. 

  317.     {
    318. 

  318.         $functions = array();
    319. 

  319.         $functions_version = array();
    320. 

  320.         $latest_version = $this->latest_version;
    321. 

  321.         $extensions = array();
    322. 

  322.         $constants = array();
    323. 

  323.         $constant_names = array();
    324. 

  324.         $udf = array();
    325. 

  325.         
    326. 

  326.         /* Check for PHP 5 stuffs */
    327. 

  327.         
    328. 

  328.         $php5_tokens = @array(
    329. 

  329.                         T_ABSTRACT => 'abstract',
    330. 

  330.                         T_CATCH => 'catch',
    331. 

  331.                         T_FINAL => 'final',
    332. 

  332.                         T_INSTANCEOF => 'instanceof',
    333. 

  333.                         T_PRIVATE => 'private',
    334. 

  334.                         T_PROTECTED => 'protected',
    335. 

  335.                         T_PUBLIC => 'public',
    336. 

  336.                         T_THROW => 'throw',
    337. 

  337.                         T_TRY => 'try',
    338. 

  338.                         T_CLONE => 'clone',
    339. 

  339.                         T_INTERFACE => 'interface',
    340. 

  340.                         T_IMPLEMENTS => 'implements',
    341. 

  341.                         );
    342. 

  342.  
    343. 

  343.         foreach ($php5_tokens as $php5_token => $value) {
    344. 

  344.             if (in_array(array($php5_token, $value), $tokens) 
    345. 

  345.                 || in_array(array($php5_token, strtoupper($value)), $tokens) 
    346. 

  346.                 || in_array(array($php5_token, ucfirst($value)), $tokens)) {
    347. 

  347.                 $constants[] = $php5_token;
    348. 

  348.                 $latest_version = '5.0.0';
    349. 

  349.                 break;
    350. 

  350.             }
    351. 

  351.         }
    352. 

  352.         
    353. 

  353.         $token_count = sizeof($tokens);
    354. 

  354.         $i = 0;
    355. 

  355.         while ($i < $token_count) {
    356. 

  356.             $found_func = true;
    357. 

  357.             if ($tokens[$i][0] == T_FUNCTION) {
    358. 

  358.                 $found_func = false;
    359. 

  359.             }
    360. 

  360.             while ($found_func == false) {
    361. 

  361.                 $i += 1;
    362. 

  362.                 if ($tokens[$i][0] == T_STRING) {
    363. 

  363.                     $found_func = true;
    364. 

  364.                     $udf[] = $tokens[$i][1];
    365. 

  365.                 }
    366. 

  366.             }
    367. 

  367.             if ($tokens[$i][0] == T_STRING) {
    368. 

  368.                 if (isset($tokens[$i + 1]) && ($tokens[$i + 1][0] == '(') && ($tokens[$i - 1][0] != T_DOUBLE_COLON) && ($tokens[$i - 1][0] != T_OBJECT_OPERATOR)) {
    369. 

  369.                     $functions[] = $tokens[$i][1];
    370. 

  370.                 }
    371. 

  371.             }
    372. 

  372.             if (in_array($tokens[$i][0],$GLOBALS['const']['tokens'])) {
    373. 

  373.                 $constants[] = $tokens[$i][0];
    374. 

  374.             }
    375. 

  375.             $i += 1;
    376. 

  376.         }
    377. 

  377.  
    378. 

  378.         $functions = array_unique($functions);
    379. 

  379.         if (isset($options['ignore_functions'])) {
    380. 

  380.             $options['ignore_functions'] = array_map("strtolower",$options['ignore_functions']);
    381. 

  381.         } else {
    382. 

  382.             $options['ignore_functions'] = array();
    383. 

  383.         }
    384. 

  384.         foreach($functions as $name) {
    385. 

  385.             if (isset($GLOBALS['funcs'][$name]) && (!in_array($name,$udf) && (!in_array($name,$options['ignore_functions'])))) {
    386. 

  386.                 if ($options['debug'] == true) {
    387. 

  387.                     $functions_version[$GLOBALS['funcs'][$name]['init']][] = array('function' => $name, 'extension' => $GLOBALS['funcs'][$name]['ext']);
    388. 

  388.                 }
    389. 

  389.                 $cmp = version_compare($latest_version,$GLOBALS['funcs'][$name]['init']);
    390. 

  390.                 if ((int)$cmp === -1) {
    391. 

  391.                     $latest_version = $GLOBALS['funcs'][$name]['init'];
    392. 

  392.                 }
    393. 

  393.                 if ((!empty($GLOBALS['funcs'][$name]['ext'])) && ($GLOBALS['funcs'][$name]['ext'] != 'ext_standard') && ($GLOBALS['funcs'][$name]['ext'] != 'zend'))  {
    394. 

  394.                     $extension = substr($GLOBALS['funcs'][$name]['ext'],4);
    395. 

  395.                     if ($extension{0} != '_') {
    396. 

  396.                         if(!in_array($extension,$extensions)) {
    397. 

  397.                             $extensions[] = $extension;
    398. 

  398.                         }
    399. 

  399.                     } else {
    400. 

  400.                         $ext = substr($extension, 1);
    401. 

  401.                         if(!in_array($extension,$extensions)) {
    402. 

  402.                             $extensions[] = $extension;
    403. 

  403.                         }
    404. 

  404.                     }
    405. 

  405.                 }
    406. 

  406.             }
    407. 

  407.         }
    408. 

  408.  
    409. 

  409.         $constants = array_unique($constants);
    410. 

  410.         foreach($constants as $constant) {
    411. 

  411.             $cmp = version_compare($latest_version,$GLOBALS['const'][$constant]['init']);
    412. 

  412.             if ((int)$cmp === -1) {
    413. 

  413.                 $latest_version = $GLOBALS['const'][$constant]['init'];
    414. 

  414.             }
    415. 

  415.             if(!in_array($GLOBALS['const'][$constant]['name'],$constant_names)) {
    416. 

  416.                 $constant_names[] = $GLOBALS['const'][$constant]['name'];
    417. 

  417.             }
    418. 

  418.         }
    419. 

  419.  
    420. 

  420.         ksort($functions_version);
    421. 

  421.  
    422. 

  422.         $functions_version['constants'] = $constant_names;
    423. 

  423.         $functions_version['extensions'] = $extensions;
    424. 

  424.         $functions_version['version'] = $latest_version;
    425.
